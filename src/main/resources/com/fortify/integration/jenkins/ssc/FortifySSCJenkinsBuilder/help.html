This step executes one or more SSC actions.

Jenkins Pipeline Example:

  stage("Fortify Results") {
    node {
      sscPerformActions(
        select: [applicationName: 'WebGoat', versionName: '5.0'],
        actions: [
          sscCreateApplicationVersionIfNotExisting(),
          sscUploadFPR(fprAntFilter: '**/*.fpr')
        ])
    }
  }
