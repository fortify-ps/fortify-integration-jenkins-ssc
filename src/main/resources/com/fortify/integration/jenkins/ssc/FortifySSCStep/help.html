This step executes one or more SSC actions.

Example:

      stage("Fortify Results") {
          node {
              sscPerformActions(applicationName: 'WebGoat', versionName: '5.0', actions: [
              	sscCreateApplicationVersionIfNotExisting(),
              	sscUploadFPR(processingTimeOutSeconds: 300)
              ])   
          }
      }
